排序。

#include <cstdio>
#include <algorithm>
using namespace std;

struct Info {
    char name[15];
    char id[15];
    int score;
} stu[105];

bool cmp(Info stu01, Info stu02) {
    return stu01.score < stu02.score;
}

int main() {
    int n; scanf("%d", &n);
    for (int i = 0; i < n; ++ i)
        scanf("%s %s %d", stu[i].name, stu[i].id, &stu[i].score);
    sort(stu, stu + n, cmp);
    printf("%s %s\n", stu[n - 1].name, stu[n - 1].id);
    printf("%s %s", stu[0].name, stu[0].id);
    return 0;
}
