使用内置包 net/http 进行 web 开发

package main

import (
    "fmt"
    "net/http"
)

func sayHello(w http.ResponseWriter, r *http.Request) {
    // Fprinln 往文件中写字符串
    _, _ = fmt.Fprintln(w, "<h1>Hello, Goland!</h1>")
    _, _ = fmt.Fprintln(w, "<h2>Hello, Goland!</h2>")
    _, _ = fmt.Fprintln(w, "<h3>Hello, Goland!</h3>")
}

func main() {
    // sayHello 这个函数必须要接收两个参数，第一个参数是 http 包里的 ResponseWriter，表示响应，返回给浏览器的内容
    // 第二个参数是一个指针，*http.Request，表示请求
    // web 开发就是一个请求对应一个响应
    http.HandleFunc("/hello", sayHello)
    // 回应与请求
    err := http.ListenAndServe(":9001", nil)
    if err != nil {
        fmt.Printf("HTTP server failed, err: %v\n", err)
        return
    }
}

可以在函数中导入文件，使代码更有条理。

package main

import (
    "fmt"
    "net/http"
)

func sayHello(w http.ResponseWriter, r *http.Request) {
    b, err = ioutil.ReadFile("./hello.txt")
    _, _ = fmt.Fprintln(w, string(b))
}

func main() {
    // sayHello 这个函数必须要接收两个参数，第一个参数是 http 包里的 ResponseWriter，表示响应，返回给浏览器的内容

    // web 开发就是一个请求对应一个响应
    http.HandleFunc("/hello", sayHello)
    // 回应与请求
    err := http.ListenAndServe(":9001", nil)
    if err != nil {
        fmt.Printf("HTTP server failed, err: %v\n", err)
        return
    }
}
