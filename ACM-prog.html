闰年

#include<iostream>
using namespace std;

//判断是否是闰年
//普通闰年:公历年份是4的倍数，且不是100的倍数为普通闰年
//世纪闰年:公历年份是整百数的，且是400的倍数是世纪闰年
bool isLeap(int year) {
    return (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);
}

int main() {
    int year;
    scanf("%d", &year);
    printf("%d\n", isLeap(year));
    cout << isLeap(year) << endl;
    return 0;
}

判断一个数是否为整数

#include<stdio.h>
const double eps = 1e-8;

int main(void) {
    double x;
    scanf("%lf", &x);
    if ((double)x - (int)x < eps)
        printf("%.6f is a positive number!\n", x);
    else
        printf("%.6f is not a positive number!\n", x);
    return 0;
}

判断一个数是否为平方数

#include <cstdio>
#include <cmath>

int main(){
    int n ;
    scanf("%d", &n) ;
    if(n == (int)sqrt(n) * (int)sqrt(n))
        printf("Yes!\n");
    else
        printf("No!\n");
    return 0;
}

程序运行时间

获得C语言程序的运行时间，调用头文件time.h，其中提供了clock()函数，可以捕捉程序运行到clock()被调用时所耗费的时间。这个时间的单位是clock tick，即“时钟打点”。同时还有一个常数CLK_TCK – 给出机器时钟每秒所走的时钟打点数。
为了获得一个函数function()的运行时间，只要在调用这个函数function()之前先调用clock()，获得一个时钟打点数C1；在这个函数function()执行完之后再调用clock()，获得另一个时钟打点数C2；两次获得的时钟打点之差(C2 - C1)就是这个函数function()运行所消耗的时钟打点数，再除以常数CLK_TCK，就得到了以秒为单位的运行时间

#include <cstdio>
#include <ctime>
int main() {
    int C1 = clock();

    //程序

    int C2 = clock();
    /*
     *printf("%d\n", C1);
     *printf("%d\n", C2);
     *printf("%d\n", CLK_TCK);
     */
    printf("%.3f\n", (double)(C2 - C1) / CLK_TCK);
    return 0;
}
