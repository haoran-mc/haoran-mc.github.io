配置json文件并调试项目

linux

使用CMake

  • task.json

    {
    // See https://go.microsoft.com/fwlink/?LinkId=733558
    // for the documentation about the tasks.json format
    "version": "2.0.0",
    "options": {
        "cwd": "${workspaceFolder}/build"  // 进入到build文件夹的绝对路径中
    },
    "tasks": [ //包含了三个小的tasks
        {
            "type": "shell",
            "label": "cmake",   // task的名字
            "command": "cmake", // 命令
            "args": [".."]      // 参数
        },
        {
            "label": "make",    //task的名字
            "group": {
                "kind": "build",
                "isDefault": true
            },
            "command": "make",  // 命令
            "args": []          // 没有参数
        },
        {
            "label": "Build",   // 这个label依赖于上面两个小的label
            "dependsOrder": "sequence", //按列出的顺序执行任务依赖项
            "dependsOn":[
                "cmake",
                "make"
            ]
        }
    ]
}

  • launch.json

    {
    // Use IntelliSense to learn about possible attributes.
    // Hover to view descriptions of existing attributes.
    // For more information, visit: https://go.microsoft.com/fwlink/?linkid=830387
    "version": "0.2.0",
    "configurations": [
        {
            "name": "g++ - 生成和调试活动文件",  // launch的一个名字
            "type": "cppdbg",                    // 不用管，用不到其他的
            "request": "launch",                 // 不用管，用不到其他的
            // "program": "${fileDirname}/${fileBasenameNoExtension}",
            // *** 代表了要调试的可执行文件的绝对路径(非常重要)
            // 要把生成的可执行文件的路径放在这里，如果用的是CMake的话，就把
            // CMake生成的可执行文件路径放在这里，如果是使用G++命令生成的话，也要放在这里
            // 在这里使用${workspaceFolder}，代表这个项目的顶层目录，可执行文件是
            // build文件夹下的my_cmake_exe
            "program": "${workspaceFolder}/build/my_cmake_exe",    
            "args": [],  // int main(int argc, char **argv)  有参数的时候才需要参数
            "stopAtEntry": false,
            "cwd": "${workspaceFolder}", //进入到当前目录
            "environment": [],
            "externalConsole": false,
            "MIMode": "gdb", //调试器的路径
            "miDebuggerPath": "/usr/bin/gdb" //gdb的路径
            "setupCommands": [
                {
                    "description": "为 gdb 启用整齐打印",
                    "text": "-enable-pretty-printing",
                    "ignoreFailures": true
                }
            ],
            // 在launch之前做一个task，也就是一个任务(没有CMake的话不需要)
            "preLaunchTask": "Build",  // 要与task文件中的label名字相同
        }
    ]
}
  • "miDebuggerPath" 一定是自己当前使用电脑的gdb.exe路径
  • "program" 一定是cmakelists.txt生成的可执行文件的路径及名字
  • "preLaunchTask" 如果要只调试当前生成的exe文件，可将此行注释。如果是先build再直接调试则不用注释。(注：build 和debug是分开的，不注释则合并在一起操作了，如果注释了，要先手动build，再debug)
不使用CMake

注意: 生成的可执行文件一定要在项目目录下，所以需要一个单独的.exe文件夹存储可执行文件

  • launch.json

    {
    "version": "0.2.0",
    "configurations": [
        {
            "name": "C++ Launch",
            "type": "cppdbg",
            "request": "launch",
            // 1. "program": "${workspaceRoot}/a.out",
            // 2. "program": "${fileDirname}/${fileBasenameNoExtension}",
            // 3. "program": "${workspaceRoot}/${fileBasenameNoExtension}",
            "program": "${workspaceRoot}/.exe/${fileBasenameNoExtension}",
            // "program": "~/haoran/work/editor/VScode/_exe/${fileBasenameNoExtension}",  //这个不可以
            "args": [],
            "stopAtEntry": true,
            "externalConsole": false,
            "cwd": "${workspaceFolder}",
            "preLaunchTask": "Compile"
            /*"windows":{
             *  "MIMode": "gdb",
             *  "miDebuggerPath": "C:\\Windows\\mingw64\\bin\\gdb.exe"
             *}
             */
        }
    ]
}

  • task.json

    {
    "version": "2.0.0",
    "options": {"cwd": "${workspaceFolder}"},
    "tasks": [
        {
            "label": "Compile",
            "type": "shell",
            "command": "g++",
            "args":[
                "-g",
                "${file}",
                "-o",
                // 1. "${workspaceRoot}/a.out"
                // 2. "${fileDirname}/${fileBasenameNoExtension}",
                // 3. "${workspaceRoot}/${fileBasenameNoExtension}"
                "${workspaceRoot}/.exe/${fileBasenameNoExtension}"

            ],
            "group": {
                "kind": "build",
                "isDefault": true
            }
        }
    ]
}
