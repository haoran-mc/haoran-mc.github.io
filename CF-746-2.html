A - Gamer Hemose

排个序选择伤害最高的两种武器。

#include <cstdio>
#include <algorithm>
using namespace std;
const int maxn = 1e3 + 5;
const int INF = 0x3f3f3f3f;
int main() {
    int t; scanf("%d", &t);
    while (t --) {
        int n, H;
        scanf("%d %d", &n, &H);
        int arr[maxn];
        for (int i = 0; i < n; ++ i)
            scanf("%d", &arr[i]);
        sort(arr, arr + n);
        int first = arr[n - 1], second = arr[n - 2];
        int cnt = H / (first + second);
        // printf("%d %d %d\n", first, second, cnt);
        if ((first + second) * cnt == H)
            printf("%d\n", 2 * cnt);
        else if ((first + second) * cnt + first >= H)
            printf("%d\n", 2 * cnt + 1);
        else
            printf("%d\n", 2 * cnt + 2);
    }
    return 0;
}

B - Hemose Shopping

如果有这样一个序列 \(b a c d e f\)，需要将它变成 \(a b c d e f\)，那么可以进行下面的操作。

\[b a c d e f \rightarrow b e c d a f \rightarrow a b c d e f\]

上面这个操作将相邻的 \(b\) 和 \(a\) 交换，参考冒泡排序，所以如果给出的序列中任意位置的字符都能交换，那么就可以排序。

只要中间的那个字符可在 \(x\) 字符内交换，那么所有的都能交换成功，但是如果中间的位置的字符本就是应该排序的字符，那么就看两侧的字符。

        int arr[maxn], sorted[maxn];
        bool f = true;
        arr[0] = -1;
        for (int i = 1; i <= n; ++ i) {
            scanf("%d", &arr[i]);
            sorted[i] = arr[i];
            if (arr[i] < arr[i - 1])
                f = false;
        }
        sort(sorted + 1, sorted + n + 1);
        if (f) {
            printf("YES\n");
            continue;
        }
        else {
            int mid = n / 2;
            if (n & 1) {
                mid ++ ;
                int l = mid, r = mid;
                while (l >= 0 && r <= n - 1 && arr[l] == sorted[l] && arr[r] == sorted[r]) {
                    mid ++ ;
                    l -- , r ++ ;
                }
            }
            else {
                mid ++ ;
                int l = mid - 1, r = mid;
                while (l >= 0 && r <= n - 1 && arr[l] == sorted[l] && arr[r] == sorted[r]) {
                    mid ++ ;
                    l -- , r ++ ;
                }
            }
            if (mid - 1 >= x)
                printf("YES\n");
            else
                printf("NO\n");
        }
    }
    return 0;
}

#include <cstdio>
#include <algorithm>
using namespace std;
const int maxn = 1e5 + 5;
const int INF = 0x3f3f3f3f;
int a[maxn], b[maxn];

int main() {
    int t; scanf("%d",&t);
    while (t -- ) {
        int n, x;
        scanf("%d %d",&n,&x);
        for(int i = 1; i <= n; ++ i) {
            scanf("%d",&b[i]);
            a[i] = b[i];
        }
        sort(a + 1, a + n + 1);
        int res = INF;
        for (int i = 1; i < n; ++ i)
            if (a[i] != b[i])
                res = min(res, max(i - 1, n - i));
        if (res < x)
            printf("NO\n");
        else
            printf("YES\n");
    }
    return 0;
}
