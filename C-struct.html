结构体中查找

#include <iostream>
using namespace std;
struct Node{
    int A;
    int B;
    MY(){}
    MY(int A,int B): A(_A), B(_B) {}
    bool operator<(const Node x)const{ //定义比较方式，这一步很重要
        return A < x.A;
    }
};

int main(){
    Node x[10];
    for(int i = 0; i < 10; i++){
        m[i] = Node(i + 1, 2 * i);
        cout << m[i].A << "  "<< m[i].B << endl;
    }
    cout << "请输入你需要查找的数字a:" << endl;
    int num;
    cin >> num;
    sort(x, x + 10);//进行二分之前需要排序
    //需要把我们查找的数封装成一个结构体才能进行查找。
    int position = lower_bound(x, x + 10, Node(num, 0)) - m;
    cout << "查到位置为：" << position << endl;
    return 0;
}

vector与结构体

#include <iostream>
using namespace std;
struct Node{
    int A;
    int B;
    MY(){}
    MY(int A,int B): A(_A), B(_B) {}
    bool operator<(const Node x)const{ //定义比较方式，这一步很重要
        return A < x.A;
    }
};

int main(){
    vector<Node> vec;
    for(int i = 0; i < 10; i++){
        vec.push_back(Node(i+1, 2*i));
        cout << vec[i].A << "  " << vec[i].B << endl;
    }
    cout << "请输入你需要查找的数字a:" << endl;
    int num;
    cin >> num;
    sort(vec.begin(), ve.end());//进行二分之前需要排序
    //需要把我们查找的数封装成一个结构体才能进行查找。
    int position = lower_bound(vec.begin(), vec.end(), Node(num, 0)) - ve.begin();
    cout << "查到位置为：" << position << endl;
    return 0;
}
