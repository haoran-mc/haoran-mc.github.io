 *   题    目：PAT 1001 害死人补偿命的(3n+1)猜想
 *   算    法：<++>
 *   描    述：<++>
 *
 ----------------------------------------------------------------*/

#include <cstdio>
int main() {
    int n; scanf("%d", &n);
    int cnt = 0;
    while (n != 1) {
        if (n & 1)
            n = (3 * n + 1) / 2;
        else
            n /= 2;
        cnt ++;
    }
    printf("%d", cnt);
    return 0;
}
