package main

import (
    "context"
    "example/20220825/grpchelloworld/proto"
    "fmt"

    "google.golang.org/grpc"
    "google.golang.org/grpc/credentials/insecure"
)

func main() {
    conn, _ := grpc.Dial("localhost:8888",
        grpc.WithTransportCredentials(insecure.NewCredentials()))
    defer conn.Close()

    client := proto.NewHelloClient(conn)
    req, _ := client.SayHello(context.Background(), &proto.Request{Msg: "client here"})
    fmt.Println(req.GetMsg())
}
