A - AB Balance

如果输入的字符串的第一个字符与最后一个字符一样，那么就一定满足题意，否则就操作一次：把第一个字符改成最后一个字符。

#include <cstdio>
#include <cstring>
const int maxn = 100 + 5;

int main() {
    int t; scanf("%d", &t);
    while (t -- ) {
        char str[maxn];
        scanf("%s", str);
        int len = strlen(str);
        if (str[0] == str[len - 1])
            printf("%s\n", str);
        else {
            str[0] = str[len - 1];
            printf("%s\n", str);
        }
    }
    return 0;
}

B - Update Files

一开始使用二进制模拟一下，如果电缆数比二进制小了，就直接除。

#include <cstdio>
#include <cmath>
#define lowbit(x) ((x) & -(x))   // lowbit(ob0100) = 4
typedef long long ll;
const ll maxn = 1e18;
ll preS[64 + 5];
ll pow2[64 + 5];
#define bug puts("<-->");
#define NEXTLINE puts("");

void work_base2() {
    int idx = 1;
    preS[0] = 1;
    for (ll i = 1; i < maxn; i *= 2, idx ++ ) {
        pow2[idx] = i;
        preS[idx] = preS[idx - 1] + i;
    }
}


int main() {
#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    int t; scanf("%d", &t);
    work_base2();
    while (t -- ) {
        ll n, k;
        scanf("%lld %lld", &n, &k);
        if (n == 1) {
            printf("%d\n", 0);
            continue;
        }
        ll tmp = k;
        int cnt = 1;
        while (k >> 1) {
            k >>= 1;
            cnt ++ ;
        }
        ll res = 0;
        if (n <= preS[cnt]) {
            for (int i = 1; i <= cnt; ++ i)
                if (preS[i] >= n) {
                    res = i;
                    break;
                }
        }
        else {
            n -= preS[cnt];
            res = cnt;
            res += n / tmp;
            if (n % tmp != 0)
                res ++ ;
        }
        printf("%lld\n", res);
    }
    return 0;
}

C - Banknotes

把较小的面值用最多次，直到最后的一个面值到不了下一个面值结束。

比如有 \(1\) 和 \(1000\)，如果仅让用 \(998\) 张钞票，那么就看这 \(998 + 1\) 张钞票能不能到达下一个面值，如果不能，那么就忽略后面所有面值的钞票，因为前面这些钞票组成的面值都有不能拼成的。

如果能够到达下一个面值，那就把当前面值用尽量多张，例如上面的例子，如果让用 \(999\) 张钞票，\(999 + 1\) 能到达下一个面值，所以面值为 \(1\) 的钞票用尽可能多：\(1000 / 1 - 1 = 999\)，由于是让你找出第一个不能构成的总面值，所以我们认为可以使用 \(1000\) 张构成的最小面值，那么还剩一张使用 \(1000\)，也就是 \(1999\)。

#include <cstdio>
int den[15];
typedef long long ll;
#define bug puts("<-->");
#define NEXTLINE puts("");

int base10(int expo) {
    int val = 1;
    for (int i = 0; i < expo; ++ i)
        val *= 10;
    return val;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    int t; scanf("%d", &t);
    int idx = 0;
    while (t -- ) {
        int n, k;
        scanf("%d %d", &n, &k);
        for (int i = 0; i < n; ++ i)
            scanf("%d", &den[i]);
        ll val = 0;
        for (int i = 0; i < n - 1 && k; ++ i) {
            // 尽量用掉 c 张面值为 base10(den[i]) 的钞票
            int c = base10(den[i + 1]) / base10(den[i]) - 1;
            // 如果能够用 c 张，那就用 c 张
            if (c < k) {
                val += c * base10(den[i]);
                k -= c;
            }
            else if (c == k) {
                val += c * base10(den[i]);
                val += 1 * base10(den[i + 1]);
                k -= c;
            }
            else {   // k 太小了
                k ++ ;
                val += k * base10(den[i]);
                k = 0;
            }
        }
        // 如果前面 n - 1 种面值尽可能多但还没有用完钞票，那就全用第 n 种
        if (k) {
            val += (ll)(k + 1) * base10(den[n - 1]);
        }
        printf("%lld\n", val);
    }
    return 0;
}
