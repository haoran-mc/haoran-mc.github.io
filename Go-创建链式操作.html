创建链式操作

package main

import "fmt"

type cli struct {
}

func NewCli() *cli {
    return &cli{}

}
func (ths *cli) Where() *cli {
    fmt.Println("where")
    return ths
}

func (ths *cli) Join() *cli {
    fmt.Println("join")
    return ths
}

func (ths *cli) Find() string {
    return "find"
}

func main()  {
    findData := NewCli().Where().Join().Find()
    fmt.Print(findData)
}

where
join
find

所谓的链式调用，实质上就是一个方法返回了一个对象，该对象还有其他方法可以继续调用。

如果是多返回值，那么就不能进行链式调用了。
