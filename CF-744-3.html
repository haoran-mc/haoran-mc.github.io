A Casimir's String Solitaire

只要输入的这个字符串中'A'，'C'字符的个数总和是'B'的个数就可以全部消掉。 

#include <cstdio>
#include <cstring>

int main() {
		int t; scanf("%d", &t);
		while (t -- ) {
				char str[100];
				scanf("%s", str);
				int cnt_ac = 0, cnt_b = 0;
				for (int i = 0; i < strlen(str); ++ i)
						if (str[i] == 'B') cnt_b ++ ;
						else cnt_ac ++ ;

				if (cnt_b == cnt_ac)
						printf("YES\n");
				else
						printf("NO\n");
		}
		return 0;
}

B Shifting Sort

插入排序，处理第\(i\)个数时，前\(i - 1\)个数都已经排好，那么将第\(i\)个数插入到前\(i - 1\)的数组中，同时记录插入的位置(j)，i，(i - j)就是左移的位数。 

#include <cstdio>
#include <vector>
using namespace std;
const int maxn = 55;
int arr[maxn];
struct Node {
		int l, r, d;
};
vector<Node> vec;

void insertSort(int num[], int n) {
		// 前i - 1个数已经排好(不是1234这种，是1357这种)，看第i个数插在哪
		for (int i = 1; i < n; ++ i) {
				int insertNum = num[i];
				int j;
				for (j = i; j >= 1 && insertNum < num[j - 1]; -- j)
						num[j] = num[j - 1];
				num[j] = insertNum;
				if (i != j)
						vec.push_back({j + 1, i + 1, i - j});
		}
}

D Productive Meeting

E1 Permutation Minimization by Deque

用一个deque模拟一下就行了 

#include <cstdio>
#include <deque>
using namespace std;
#define NEXTLINE puts("");

int main() {
		int t; scanf("%d", &t);
		while (t --) {
				deque<int> deq;
				int n; scanf("%d", &n);
				for (int i = 0; i < n; ++ i) {
						int _; scanf("%d", &_);
						if (deq.size() == 0)
								deq.push_front(_);
						else if (_ < deq.front())
								deq.push_front(_);
						else
								deq.push_back(_);
				}
				while (deq.size()) {
						printf("%d ", deq.front());
						deq.pop_front();
				}
				NEXTLINE;
		}
		return 0;
}
