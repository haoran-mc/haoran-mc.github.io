n = eval(input())

for i in range(n):
        str = "_" + input()   # 下标从1开始
        pos_P = -1
        pos_T = -1
        is_valid = True
        for j in range(1, len(str)):
                if str[j] == 'P' and pos_P == -1:
                        pos_P = j
                elif str[j] == 'T' and pos_T == -1:
                        pos_T = j
                elif str[j] == 'P' and pos_P != -1:
                        is_valid = False
                        break
                elif str[j] == 'T' and pos_T != -1:
                        is_valid = False
                        break
                elif str[j] != 'A':
                        is_valid = False
                        break
                cnt01 = pos_P - 1
                cnt02 = pos_T - pos_P - 1
                cnt03 = len(str) - pos_T - 1
        if is_valid != True:
                print("NO")
        elif cnt02 == 0:
                print("NO")
        elif cnt01 == cnt03 and cnt02 == 1:
                print("YES")
        elif cnt03 != cnt01 * cnt02:
                print("NO")
        else:
                print("YES")
