HDU 2844 Coins

题目大意

有 \(n\) 种面值的硬币，每枚硬币的面值是 \(A_i\)，数量是 \(C_i\)，选出一部分硬币，其面值总和为 \(sum\)，则不超过 \(m\) 的 \(sum\) 有多少种。

看一组数据：
3 10
1 2 4 2 1 1

一共有 \(3\) 种面值的硬币，分别是：面值为 \(1\) 的硬币有 \(2\) 个；面值为 \(2\) 的硬币有 \(1\) 个；面值为 \(4\) 的硬币有 \(1\) 个。

所以它们能组成的面值总和有：

  1. 1 = 1
  2. 2 = 2
  3. 3 = 1 + 2
  4. 4 = 4
  5. 5 = 1 + 4
  6. 6 = 2 + 4
  7. 7 = 1 + 2 + 4
  8. 8 = 1 + 1 + 2 + 4
  9. 面值总和不能为 \(9\) 或 \(10\)。

使用多重背包的方法解决

转化为有 \(m\) 种不同容量的背包，容量分别为 \(1, 2, 3, \dots, m\)，现在给你 \(n\) 个体积是 \(A_i\)，价值是 \(1\)，数量是 \(C_i\) 的物品来装背包，问能恰好装满的背包数量。

这里的价值可以随便取大于 \(0\) 的值，因为我们把面值转化为了体积，无论是总面值还是单个硬币的面值，体积才是本题的关键。最后判断 \(dp[i]\) 是否大于 \(0\) 就知道这个面值能否被组成。

这里是恰好背包，初始化应该是 \(dp[0] = 0, dp[i] = -INF\ (i \neq 0)\)。

AC 代码

#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
const int maxn = 100 + 5, maxm = 1e5 + 5;
const int INF = 0x3f3f3f3f;
int n, m;
int dp[maxm];
int w[maxn], s[maxn];

struct Bone {
    int v, w;
} bone[maxn * 10];

int main() {
#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    while (scanf("%d %d", &n, &m) && (n || m)) {
        for (int i = 1; i <= n; ++ i)
            scanf("%d", &w[i]);   // 每种硬币的价值
        for (int i = 1; i <= n; ++ i)
            scanf("%d", &s[i]);   // 每种硬币的数量
        int idx = 1;
        for (int i = 1; i <= n; ++ i) {
            int b = 1;
            while (s[i] - b > 0) {
                s[i] -= b;
                bone[idx].v = b * w[i];
                bone[idx].w = 1;
                idx ++ ;
                b *= 2;
            }
            if (s[i]) {
                bone[idx].v = s[i] * w[i];
                bone[idx].w = 1;
                idx ++ ;
            }
        }
        memset(dp, -INF, sizeof dp);
        dp[0] = 0;
        for (int i = 1; i < idx; ++ i)
            for (int j = m; j >= bone[i].v; -- j)
                dp[j] = max(dp[j], dp[j - bone[i].v] + bone[i].w);
        /*
         * for (int i = 0; i <= m; ++ i)
         *     printf("-- %d\n", dp[i]);
         */
        int sum = 0;
        for (int i = 1; i <= m; ++ i)
            if (dp[i] > 0)
                sum ++ ;
        printf("%d\n", sum);
    }
    return 0;
}
