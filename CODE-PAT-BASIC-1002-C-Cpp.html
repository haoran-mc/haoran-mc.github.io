虽然我们得到了123这个数字，但是不能从前往后判断输出，所以将123每个位上的数字从后往前放在栈中，然后判断输出。

#include <cstdio>
#include <cstring>

int get_sum(char* str) {
    int len = strlen(str);
    int sum = 0;
    for (int i = 0; i < len; ++i)
        sum += str[i] - '0';
    return sum;
}

int main() {
    char output[10][10] = {"ling", "yi", "er", "san", "si", "wu", "liu", "qi", "ba", "jiu"};
    char str[105]; scanf("%s", str);
    int sum = get_sum(str);
    int stk[10], tt = 0;
    if (!sum)
        printf("ling\n");
    while (sum) {
        stk[++tt] = sum % 10;
        sum /= 10;
    }
    while (tt) {
        if (tt != 1)
            printf("%s ", output[stk[tt--]]);
        else
            printf("%s\n", output[stk[tt--]]);
    }
    return 0;
}
