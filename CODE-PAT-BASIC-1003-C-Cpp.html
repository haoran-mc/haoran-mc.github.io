条件3是指P前面的A的数量乘以PT中间的A的数量等于T后面A的数量。

#include <cstdio>
#include <cstring>
const int maxn = 100 + 5;
char str[maxn];
char text[4] = "PAT";

int main() {
    int n; scanf("%d", &n);
    for (int i = 0; i < n; ++ i)  {
        scanf("%s", str + 1);
        int len = strlen(str + 1);
        int pos_P = -1, pos_T = -1;
        bool is_valid = true;
        for (int i = 1; str[i]; ++ i) {
            if (pos_P == -1 && str[i] == 'P')
                pos_P = i;
            else if (pos_T == -1 && str[i] == 'T')
                pos_T = i;
            else if (pos_P != -1 && str[i] == 'P') {
                is_valid = false;
                break;
            }
            else if (pos_T != -1 && str[i] == 'T') {
                is_valid = false;
                break;
            }
            else if (str[i] != 'A') {
                is_valid = false;
                break;
            }
        }
        int cnt01 = pos_P - 1;            // P 前面的字符 A 的个数
        int cnt02 = pos_T - pos_P - 1;    // P、T之间的字符 A 的个数
        int cnt03 = len - pos_T;          // T 之后的字符 A 的个数
        if (!is_valid)   // 有两个及以上的 P 或 T，或者有其它字符
            puts("NO");
        else if (cnt02 == 0)   // P、T 之间一定要有 A
            puts("NO");
        else if (cnt01 == cnt03 && cnt02 == 1)   // 满足题目条件2
            puts("YES");
        else if (cnt03 != cnt02 * cnt01)   // P 前面的字符数 != P、T 之间的字符数 * T 后面的字符数
            puts("NO");
        else
            puts("YES");
    }
    return 0;
}
