搭建副本集

一、创建数据目录

mkdir -p ~/.mongo/repl/data01
mkdir -p ~/.mongo/repl/data02
mkdir -p ~/.mongo/repl/data03

二、搭建副本集

mongod --port 27017 --dbpath /home/haoran/.mongo/repl/data01 --bind_ip 0.0.0.0 --replSet myreplace/[localhost:27018, localhost:27019]

mongod --port 27018 --dbpath /home/haoran/.mongo/repl/data02 --bind_ip 0.0.0.0 --replSet myreplace/[localhost:27019, localhost:27017]

mongod --port 27019 --dbpath /home/haoran/.mongo/repl/data03 --bind_ip 0.0.0.0 --replSet myreplace/[localhost:27017, localhost:27018]

# --replSet 副本集 myreplace 副本集名称/集群中其他节点的主机和端口

三、配置副本集，连接任意节点

> use admin
  • 初始化副本集
> var config = {
    _id: "myreplace",
    members: [
        {_id: 0, host: "localhost: 27017"},
        {_id: 1, host: "localhost: 27018"},
        {_id: 2, host: "localhost: 27019"}
    ]
}

> rs.initiate(config)   # 初始化配置
  • 设置客户端临时可以访问
> rs.slaveOK()
> rs.secondaryOK()
