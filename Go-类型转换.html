类型转换

string 到 int

int, err: = strconv.Atoi(string)

string 到 int64

int64, err := strconv.ParseInt(string, 10, 64)

int 到 string

string := strconv.Itoa(int) 等价于 string := strconv.FormatInt(int64(int), 10)

int64 到 string

string := strconv.FormatInt(int64, 10)

int 到 int64

int64_ := int64(1234)

float 到 string

string := strconv.FormatFloat(float32, 'E', -1, 32) string := strconv.FormatFloat(float64, 'E', -1, 64)

string 到 float

float, err := strconv.ParseFloat(string, 64) float, err := strconv.ParseFloat(string, 32)
