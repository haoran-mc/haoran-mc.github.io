def get_sum(str):
    sum = 0
    for i in range(len(str)):
        sum += (int)(str[i])
    return sum


output = ["ling", "yi", "er", "san", "si", "wu", "liu", "qi", "ba", "jiu"]
str = input()
sum = get_sum(str)


stk = [0] * 10
tt = 0


if not sum:
    print("ling\n")
while sum:
    tt += 1
    stk[tt] = sum % 10
    sum //= 10
while tt:
    if tt != 1:
        print("%s " % output[stk[tt]], end="")
    else:
        print("%s" % output[stk[tt]])
        tt -= 1
