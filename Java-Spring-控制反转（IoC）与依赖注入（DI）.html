依赖注入（DI）

构造器注入

Set 注入

<bean id="address" class="com.meta.model.Address">
  <property name="address" value="天健19栋"/>
</bean>

<bean id="student" class="com.meta.model.Student">
  <property name="name" value="ogmc"/>            <!-- 普通注入 -->
  <property name="address" ref="address"/>        <!-- 引用注入 -->
  <property name="books">                         <!-- 数组注入 -->
    <array>
      <value>水浒传</value>
      <value>西游记</value>
      <value>红楼梦</value>
      <value>三国演义</value>
    </array>
  </property>
  <property name="hobbies">                       <!-- list 注入 -->
    <list>
      <value>打游戏</value>
      <value>动漫</value>
      <value>看代码</value>
    </list>
  </property>
  <property name="card">                          <!-- map 注入 -->
    <map>
      <entry key="身份证" value="******************"/>
      <entry key="银行卡" value="******************"/>
      <entry key="学生证" value="**********"/>
    </map>
  </property>
  <property name="games">                         <!-- set 注入 -->
    <set>
      <value>yuanshen</value>
      <value>lol</value>
    </set>
  </property>
  <property name="job">                           <!-- null 注入 -->
    <null/>
  </property>
  <property name="info">                          <!-- properties 注入 -->
    <props>
      <prop key="学号">**********</prop>
      <prop key="性别">男</prop>
    </props>
  </property>
</bean>

p 命名空间注入

<beans xmlns:p="http://www.springframework.org/schema/p">

<!-- 需要有无参构造方法 -->
<bean id="teacher" class="com.meta.model.Teacher" p:name="NEXTLINE" p:age="20"/>

c 命名空间注入

<beans xmlns:c="http://www.springframework.org/schema/c">

<!-- 需要有有参构造方法 -->
<bean id="teacher" class="com.meta.model.Teacher" c:name="NEXTLINE" c:age="20"/>
