A - Consecutive Sum Riddle

从 \(- n + 1\) 加到 \(n\)。

#include <cstdio>
typedef long long ll;

int main() {
    int t; scanf("%d", &t);
    while (t -- ) {
        ll n; scanf("%lld", &n);
        printf("%lld %lld\n", 1 - n, n);
    }
    return 0;
}

B - Special Numbers

使用二进制解决。

#include <cstdio>
#include <vector>
#include <cmath>
using namespace std;
typedef long long ll;
vector<int> vec;
const int MOD = 1e9 + 7;
#define NEXTLINE puts("");

void base10To2(ll k) {
    while (k) {
        vec.push_back(k % 2);
        k /= 2;
    }
}

ll binPow(ll base, ll expo) {
    ll res = 1;
    while (expo) {
        if (expo & 1)
            res = (1LL * res * base) % MOD;
        base = (1LL * base * base) % MOD;
        //也就是base分别是2, 4, 8, 16, ...
        expo >>= 1;
    }
    return res % MOD;
}

int main() {
    int t; scanf("%d", &t);
    while (t -- ) {
        ll n, k;
        scanf("%lld %lld", &n, &k);
        base10To2(k);
        ll res = 0;
        for (int i = 0; i < vec.size(); ++ i) {
            if (vec[i])
                res += binPow(n, (ll)i);
            res %= MOD;
        }
        printf("%lld\n", res);
        vec.clear();
    }
    return 0;
}

C - Make Them Equal

实际上只要看后半段，如果字符串中有字符 \(c\) 那么就选这个位置。否则就选择 \(n - 1, n\) 两个位置。

#include <cstdio>
#define bug printf("<-->\n");
const int maxn = 3e5 + 5;
int main() {
    int t; scanf("%d", &t);
    while (t -- ) {
        int n;
        char ch[2];
        scanf("%d %s", &n, ch);
        char str[maxn];
        scanf("%s", str);
        bool f = false;
        for (int i = 0; i < n; ++ i)
            if (str[i] != *ch) {
                f = true;
                break;
            }
        if (!f) {
            printf("0\n");
            continue;
        }
        f = false;
        for (int i = n / 2; i < n; ++ i)
            if (str[i] == *ch) {
                f = true;
                printf("%d\n%d\n", 1, i + 1);
                break;
            }
        if (!f)
            printf("%d\n%d %d\n", 2, n - 1, n);
    }
    return 0;
}

E1 - Rubik's Cube Coloring(easy version)

推公式。

#include <cstdio>
typedef long long ll;
const int MOD = 1e9 + 7;

ll work(ll base) {
    base %= MOD;
    return base * base % MOD;
}

int main() {
    int k; scanf("%d", &k);
    ll idx = k - 1, base = 1;
    while (idx -- )
        base = work(base * 4);
    base = base * 6 % MOD;
    printf("%lld\n", base);
    return 0;
}
