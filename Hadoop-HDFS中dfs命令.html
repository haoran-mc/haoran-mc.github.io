  1. hdfs dfs -help：查看帮助

  2. hdfs dfs -mkdir：创建一个文件夹

    $ hdfs dfs -mkdir /test

  3. hdfs dfs -put：将本地文件上传到 hdfs 中

    $ hdfs dfs -put 01.txt /test/

  4. hdfs dfs -ls：列出路径下的所有文件和文件夹

    $ hdfs dfs -ls /test/

  5. hdfs dfs -cat：查看文件中的内容

    $ hdfs dfs -cat /test/01.txt

  6. hdfs dfs -text：列出文件中的内容

    hdfs dfs -text /test/01.txt

  7. hdfs dfs -touchz：创建一个空文件

    hdfs dfs -touchz /text.txt

  8. hdfs dfs -ls -R：递归列出路径下的文件夹和文件

    hdfs dfs -ls -R /

  9. hdfs dfs -mv A B：将文件从 A 移动到 B

    hdfs dfs -mv /test/01.txt /test

  10. hdfs dfs -cp A B：将 A 处的文件复制一份到 B 处

    hdfs dfs -cp /test/01.txt /

  11. hdfs dfs -moveFromLocal A B：将文件从本地移动到 hdfs 系统中（本地文件会删除）

    hdfs dfs -moveFromLocal /home/haoran/01.txt /test

  12. hdfs dfs -rm：删除文件

    hdfs dfs -rm /test/01.txt

  13. hdfs dfs -rmdir：删除文件夹

    hdfs dfs -rmdir /test/test_dir

  14. hdfs dfs -get：将文件从 hdfs 系统移动到本地

    hdfs dfs -get /test/01.txt /home/haoran/

  15. hadoop fs -rm -r -skipTrash：强制删除 hdfs 非空文件夹

    hadoop fs -rm -r -skipTrash /dir
  16. hdfs getconf -confKey fs.default.name：获取 hdfs 链接地址以及端口号
