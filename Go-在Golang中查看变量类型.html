Table of Contents

fmt

package main

import "fmt"

func main() {
    str := "Hello World!"
    fmt.Println("%T", str)
    // 返回 string
}

reflect

package main

import (
    "fmt"
    "reflect"
)

func main() {
    str := "Hello world"
    fmt.Println(reflect.TypeOf(str))
    // 返回 string
}

type switch

package main

import "fmt"

var str interface{}
str = "Hello world"

switch str.(type) {
case string:
        fmt.Println("string")
case int:
        fmt.Println("int")
default:
        fmt.Println("other)
}
// 返回 string
