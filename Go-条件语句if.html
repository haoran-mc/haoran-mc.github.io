if

package main

import "fmt"

func main() {
    if 1 > 0 {
        fmt.Println("1 > 0")
    }

    if 1 < 0 {
        fmt.Println("1 < 0")
    }
}

if else

package main

import "fmt"

func main() {
    n := 2

    if n == 0 {
        fmt.Println("n = 0")
    } else if n == 1 {
        fmt.Println("n = 1")
    } else if n == 2 {
        fmt.Println("n = 2")
    } else if n == 3 {
        fmt.Println("n = 3")
    }
}

if 嵌套

package main

import "fmt"

func main() {
    a := 1
    b := 2
    if a == 1 {
        if b == 2 {
            fmt.Println("a = 1, b = 2")
        }
    }
}
